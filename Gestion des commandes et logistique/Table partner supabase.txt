CREATE TABLE delivery_partners (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL, -- Jumia Logistics, DHL, etc.
  api_key VARCHAR(255), -- Clé API pour l'intégration
  api_endpoint VARCHAR(255), -- URL de l'API du partenaire
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
